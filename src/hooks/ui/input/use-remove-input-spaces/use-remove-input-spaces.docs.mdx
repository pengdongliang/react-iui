import { useRemoveInputSpaces } from './use-remove-input-spaces';

## A React Hook to increment a count

Increments the state of `count` by 1 each time `increment` is called

### Component usage

```tsx
const removeInputSpacesEvent = useRemoveInputSpaces({})

return (
  <Form form={form}>
    <Form.Item name="input">
      <Input {...removeInputSpacesEvent} />
    </Form.Item>
  </Form>
)
```
